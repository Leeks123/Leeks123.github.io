# callback vs Promise vs async,await

1초마다 숫자 증가

# callback

function increase(number, callback) {
  setTimeout(() => {
    const result = number + 10;
    if (callback) {
      callback(result);
    }
  }, 1000);
}

console.log("작업 시작");
increase(0, (result) => {
  console.log(result);
  increase(result, (result) => {
    console.log(result);
    increase(result, (result) => {
      console.log(result);
      increase(result, (result) => {
        console.log(result);
        console.log("작업 완료");
      });
    });
  });
});
  • 가독성이 떨어짐 (콜백 지옥)
  • 복잡도 증가
  • 에러 처리 곤란

    try...catch로 처리하면 되지 않나?
    콜백 내부에서 발생한 예외는 호출자 방향으로 전파된다. 그러나 콜백함수가 일련의 과정을 거쳐서 콜백큐로 들어가는 과정 중에 콜백을 호출한 함수는 콜스택에서 제거된다. 즉 호출자가 사라져 버린다. 호출자가 사라져 버린 상황에서 에러는 캐치되지 않게된다.

# Promise

A promise is an object that may produce a single value some time in the future

프로미스는 자바스크립트 비동기 처리에 사용되는 객체.

pending,fulfilled, rejected의 3가지 상태를 가진다.

then 메서드를 통한 후속 처리, catch 메서드를 통한 에러 처리가 가능하다.

function increase(number) {
  const promise = new Promise((resolve, reject) => {
    setTimeout(() => {
      const result = number + 10;
      if (result > 50) {
        const e = new Error("bigger than 50");
        return reject(e);
      }
      resolve(result);
    }, 1000);
  });
  return promise;
}

increase(0)
  .then((number) => {
    console.log(number);
    return increase(number);
  })
  .then((number) => {
    console.log(number);
    return increase(number);
  })
  .then((number) => {
    console.log(number);
    return increase(number);
  })
  .then((number) => {
    console.log(number);
    return increase(number);
  })
  .then((number) => {
    console.log(number);
    return increase(number);
  })
  .catch((e) => {
    console.log(e);
  });

# async/await

함수 앞부분에 async, 함수 내부의 promise 앞부분에 await 키워드를 사용한다.

function increase(number, callback) {
  const promise = new Promise((resolve, reject) => {
    setTimeout(() => {
      const result = number + 10;
      if (result > 50) {
        const e = new Error("big");
        return reject(e);
      }
      resolve(result);
    }, 1000);
  });
  return promise;
}

async function runTasks() {
  try {
    let result = await increase(0);
    console.log(result);
    result = await increase(result);
    console.log(result);
    result = await increase(result);
    console.log(result);
    result = await increase(result);
    console.log(result);
    result = await increase(result);
    console.log(result);
    result = await increase(result);
    console.log(result);
  } catch (e) {
    console.log(e);
  }
}

runTasks();

# 프로미스 메서드

# Promise.all([])

const p1 = new Promise((resolve, reject) => {
  setTimeout(() => resolve("p1"), 1000);
});
const p2 = new Promise((resolve, reject) => {
  setTimeout(() => resolve("p2"), 2000);
});
const p3 = new Promise((resolve, reject) => {
  setTimeout(() => resolve("p3"), 3000);
});

Promise.all([p1, p2, p3]).then((msgs) => console.log(msgs));

// 3초 후
// ["p1", "p2", "p3"]

배열로 넘겨지는 모든 프로미스들이 resolve되어야 메인 프로미스가 진행된다. resolve 반환 값들로 이뤄진 배열을 받는다.

거절되면 처음 거절된 프로미스의 거절 사유를 받는다.

# Promise.allSettled([])

const p1 = new Promise((resolve, reject) => {
  setTimeout(() => reject("p1 reject"), 1000);
});
const p2 = new Promise((resolve, reject) => {
  setTimeout(() => resolve("p2"), 2000);
});
const p3 = new Promise((resolve, reject) => {
  setTimeout(() => resolve("p3"), 3000);
});

Promise.allSettled([p1, p2, p3]).then((data) => console.log(data));

// 3초 후
// [
//    {status: "rejected", reason: "p1 reject"}
//    {status: "fulfilled", value: "p2"}
//    {status: "fulfilled", value: "p3"}
// ]

배열로 넘겨진 프로미스들의 이행, 거절에 상관없이 귀결된 프로미스들의 배열로 반환한다.

# Promise.race([])

const p1 = new Promise((resolve, reject) => {
  setTimeout(() => resolve("p1"), 1000);
});
const p2 = new Promise((resolve, reject) => {
  setTimeout(() => resolve("p2"), 2000);
});
const p3 = new Promise((resolve, reject) => {
  setTimeout(() => resolve("p3"), 3000);
});

Promise.race([p1, p2, p3]).then((msg) => console.log(msg));

// 1초 후
// p1

가장 먼저 완료된 프로미스의 결과 값을 반환한다.
