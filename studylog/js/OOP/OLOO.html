# OLOO 패턴

const Foo = {
  init(name) {
    this.name = name;
    return this;
  },
  myName() {
    return this.name;
  },
};

const Bar = (function() {
  let privateVar = "private Variable";
  return {
    init(name, label) {
      this.name = name;
      this.label = label;
      return this;
    },
    myLabel() {
      return this.label;
    },
    getPrivateVar() {
      return privateVar;
    },
  };
})();

Object.setPrototypeOf(Bar, Foo);

const a = Object.create(Bar).init("a", "Label A");

위 코드는 [You Don't Know JS]의 저자 카일 심슨이 제안하는 OLOO패턴이다.

  • 이 패턴은 constructor가 따로 없기 때문에 init함수를 따로 만들어야 한다.
  • private 프로퍼티를 가지려면 IIFE형태로 클로져를 활용해야 한다.

클래스 방식의 코드는 class 키워드를 활용하면 코드는 깔끔하게 유지할 수 있지만, 실제 자바스크립트 내부적으로 복잡한 구성을 가진다. ( constructor, prototype, __proto__등이 복작하게 연결되어 동작한다.)

OLOO 패턴에서는 객체의 프로토타입 체인으로만 상속관계가 이뤄지기 때문에 단순하고 깔끔한 구조를 가진다.
