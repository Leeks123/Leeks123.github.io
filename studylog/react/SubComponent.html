# SubComponent

서브컴포넌트는 하나의 컴포넌트 안에 존재하는 그룹 컴포넌트를 말한다.
서브컴포넌트 방식으로 구현함으로써 같은 UI를 렌더링하더라도, 코드의 가독성, 컴포넌트의 재사용성이 증가한다.

# Example

const Card = ({ children }) => {
  let subComponentList = Object.keys(Card);

  let subComponents = subComponentList.map((key) => {
    return React.Children.map(children, (child) =>
      child.type.name === key ? child : null
    );
  });

  return (
    <>
      <div className="card">{subComponents.map((component) => component)}</div>
    </>
  );
};

const Header = (props) => <div className="card-header">{props.children}</div>;
Card.Header = Header;

const Body = (props) => <div className="card-body">{props.children}</div>;
Card.Body = Body;

const Footer = (props) => <div className="card-footer">{props.children}</div>;
Card.Footer = Footer;

export default Card;

서브 컴포넌트에 대한 아이디어는 객체 프로퍼티에 대한 관점이다.

컴포넌트도 결국 자바스크립트 함수이자 객체이다. 객체에는 프로퍼티가 추가 가능하다.
하나의 컴포넌트 내부에 프로퍼티로써 서브 컴포넌트들을 추가하는 것이다.

// ...
Card.Header = Header;
Card.Body = Body;
Card.Footer = Footer;

객체의 프로퍼티들은 Object.keys 함수를 통해 프로퍼티 이름들을 배열화 시킬 수 있다.

let subComponentList = Object.keys(Card);

subComponentList로부터 렌더링할 subComponents를 뽑아낸다. 이때 React.Children api를 활용한다.

React.Children api는 props.children를 다루는 다음의 유틸함수들을 제공한다.

  • React.Children.map
  • React.Children.forEach
  • React.Children.count
  • React.Children.only
  • React.Children.toArray
let subComponents = subComponentList.map((key) => {
  return React.Children.map(children, (child) =>
    child.type.name === key ? child : null
  );
});
